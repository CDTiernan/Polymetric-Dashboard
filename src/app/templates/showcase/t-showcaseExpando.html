The content of the smp-expando-content and the smp-expando-toggle can be anything!

HTML

  
    <smp-expando open="true" target="ExpandoContentID">
      <div smp-expando-toggle>
        <any> ... </any>
      </div>
      <div smp-expando-content="ExpandoContentID">
        <div class="contentContainer">
          <any> ... </any>
        </div>
      </div>
    </smp-expando>
